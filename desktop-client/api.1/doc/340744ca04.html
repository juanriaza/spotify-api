api/offline

Functionality for dealing with offline syncing of the user's music.
Since:
  • 1.16.0
Source:
  • offline.js, line 17

Example

require(['$api/models', '$api/offline#Offline'], function(models, Offline) {
  var offline = Offline.forCurrentUser();
  var playlist = models.Playlist.fromURI('spotify:user:andreas.blixt:playlist:1itC2kKqIqun8OU05J5G8v');
  offline.enableSync(playlist);
});
