Module: views/tabbar

views/tabbar

A tab bar that can be used either for the whole app, or smaller sections. If you want a main tab bar for the whole app, you should use the UI class instead, to make it easier.
Source:
  • tabbar.js, line 26
See:
  • module:views/ui

Example

require(['$views/tabbar#TabBar'], function(TabBar) {

  var tabBar = TabBar.withTabs([
    {id: 'overview', name: 'Overview', active: true},
    {id: 'followers', name: 'Followers'},
    {id: 'following', name: 'Following'}
  ]);
  tabBar.addToDom(document.body, 'prepend');

});
